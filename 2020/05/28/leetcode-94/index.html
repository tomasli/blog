题目：
来自探索->栈和深度优先搜索

给定一个二叉树，返回它的中序 遍历。

要求迭代就迭代，一下就做完了。估计难点就是中序迭代时候，父节点如何安放的

class Solution(object):
    def inorderTraversal(self, root):
        """
        :type root: TreeNode
        :rtype: List[int]
        """
        if not root:
            return []
        res = []
        stack = [root]
        while stack:
            node = stack.pop()
            if type(node) != TreeNode:
                res.append(node)
                continue

            if node.right:
                stack.append(node.right)
            stack.append(node.val)
            if node.left:
                stack.append(node.left)

        return res
